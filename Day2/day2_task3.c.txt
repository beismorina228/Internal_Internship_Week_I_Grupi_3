#include <stdio.h>

int main() {
    int day;

    printf("Shkruaj nje numer nga 1 deri ne 7: ");

    // Kontroll nëse është dhënë input i vlefshëm numerik
    if (scanf("%d", &day) != 1) {
        printf("Input i pavlefshem! Ju lutem vendosni nje numer.\n");
        return 0;
    }

    switch(day) {
        case 1:
            printf("E hene - Dite pune\n");
            break;
        case 2:
            printf("E marte - Dite pune\n");
            break;
        case 3:
            printf("E merkure - Dite pune\n");
            break;
        case 4:
            printf("E enjte - Dite pune\n");
            break;
        case 5:
            printf("E premte - Dite pune\n");
            break;
        case 6:
            printf("E shtune - Fundjave\n");
            break;
        case 7:
            printf("E diel - Fundjave\n");
            break;
        default:
            printf("Input i pavlefshem! Vendos nje numer nga 1 deri ne 7.\n");
    }

    return 0;
}
