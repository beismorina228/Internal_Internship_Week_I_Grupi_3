#include <stdio.h>

int main() {
    int n, choice;

    // marrja e input-it
    printf("Shkruaj nje numer nga 1 deri ne 12: ");
    scanf("%d", &n);

    // validimi i input-it
    if (n < 1 || n > 12) {
        printf("Input i pavlefshem! Ju lutem shkruani nje numer nga 1 deri ne 12.\n");
        return 0;
    }

    // zgjedhja e pattern-it
    printf("Zgjidh pattern-in:\n");
    printf("1. Yje (*)\n");
    printf("2. Numra\n");
    printf("Zgjedhja juaj: ");
    scanf("%d", &choice);

    // pattern me yje
    if (choice == 1) {
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= i; j++) {
                printf("* ");
            }
            printf("\n");
        }
    }

    // pattern me numra
    else if (choice == 2) {
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= i; j++) {
                printf("%d ", j);
            }
            printf("\n");
        }
    }

    // zgjedhje e gabuar
    else {
        printf("Zgjedhje e pavlefshme!\n");
    }

    return 0;
}
